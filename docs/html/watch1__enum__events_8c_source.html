55 // declare the states:
56 //
57 // all of the HSM_STATE* macros take:
58 // 1. the state being declared;
59 // 2. the state's parent state;
60 // 3. an initial child state to move into first.
61 //
62 // in this example:
63 // 1. "ActiveState" is the root of this particular machine;
64 // 2. its parent state is the system state "HsmTopState",
65 // 3. it lists 'StoppedState' as its last parameter inorder to move to that state first.
66 //
67 // "StoppedState" and "RunningState" are also states,
68 // they are siblings and share the parent state: "ActiveState"
69 //
70 // the only difference b/t HSM_STATE_ENTER and HSM_STATE is that HSM_ENTER
71 // declares a callback that gets called whenever the state gets entered...
72 //
73 HSM_STATE_ENTER( ActiveState, HsmTopState, StoppedState );
74  HSM_STATE( StoppedState, ActiveState, 0 );
75  HSM_STATE( RunningState, ActiveState, 0 );

91 //---------------------------------------------------------------------------
92 // this is the event handling function for active state
93 //
94 // the convention <StateName>Event is used by the HSM_STATE macros to indicate the handler function
95 //
96 hsm_state ActiveStateEvent( hsm_machine hsm, hsm_context ctx, const WatchEvent* evt )
97 {
98  // by default this function does nothing....
99  hsm_state ret=NULL;
100  switch ( evt->evt ) {
101  // but, whenever the reset button is pressed...
102  case WATCH_RESET_PRESSED:
103  // it transitions to itself.
104  ret= ActiveState();
105  // self-transitions are used by statecharts to trigger exit and re-entry to the same state.
106  // that means we we'll soon get our ActiveStateEnter function called
107  // we will reset our watch timer, and b/c the macros above designated "StoppedState" as
108  // Active's first state, the statemachine will move to "Stopped"
109  break;
110  }
111  return ret;
112 }

135 // event handler for the running state
136 //
137 hsm_state RunningStateEvent( hsm_machine hsm, hsm_context ctx, const WatchEvent* evt )
138 {
139  // by default this function does nothing....
140  hsm_state ret=NULL;
141  switch (evt->evt) {
142  // but, when the 'toggle' button gets pressed....
143  case WATCH_TOGGLE_PRESSED:
144  // transition back to the stopped state
145  ret= StoppedState();
146  break;
147  // also, when a 'tick' is sent, update our timer
148  case WATCH_TICK:
149  {
150  // our context is the watch object
151  Watch* watch=((WatchContext*)ctx)->watch;
152  // our event is the tick event
153  TickEvent* tick= (TickEvent*)evt;
154  // tick by this much time
155  TickTime ( watch, tick->time );
156  // print the total time
157  printf("%d,", watch->elapsed_time );
158  // indicate to the statemachine that we've take care of this event
159  // this stops the event from being sent to our parent
160  ret= HsmStateHandled();
161  }
162  break;
163  }
164  return ret;

201  // while the statemachine is still running
202  while ( HsmIsRunning( hsm ) ) {
203  // get a key from the keyboard
204  int ch= PlatformGetKey();
205  // turn a '1' into the reset button,
206  // turn a '2' into the toggle button
207  WatchEvents events[]= { WATCH_RESET_PRESSED, WATCH_TOGGLE_PRESSED };
208  int index= ch-'1';
209  if ((index >=0) && (index < sizeof(events)/sizeof(WatchEvents))) {
210  // send the event to the state machine
211  // one of the handler functions will get called as a result
212  // ( for example: StoppedStateEvent. )
213  WatchEvent evt= { events[index] };
214  HsmProcessEvent( hsm, &evt );
215  printf(".");
216  }
217  else {
218  // send a tick to the machine
219  // we send this regardless of whether we think the watch is running or stopped.
220  // the logic of the statemachine internally knows which states are currently active
221  // and sends the event to the appropriate state. noting (in the code above) that only
222  // RunningStateEvent has code does anything when it hears the tick event.
223  TickEvent tick= { WATCH_TICK, 1 };
224  HsmProcessEvent( hsm, &tick.evt );
225  PlatformSleep(500);
226  }
227  };
