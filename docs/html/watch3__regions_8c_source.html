204 hsm_state ActiveState3Event( hsm_machine hsm, hsm_context ctx, const WatchEvent* evt )
205 {
206  // by default this function does nothing....
207  hsm_state ret=NULL;
208  switch ( evt->evt ) {
209  // but, whenever the reset button is pressed...
210  case WATCH_RESET_PRESSED:
211  // it transitions to itself.
212  ret= ActiveState3();
213  // self-transitions are used by statecharts to trigger exit and re-entry to the same state.
214  // that means we we'll soon get our ActiveStateEnter function called
215  // we will reset our watch timer, and b/c the macros above designated "StoppedState" as
216  // Active's first state, the statemachine will move to "Stopped"
217  break;
218  }

245 hsm_state RunningState3Event( hsm_machine hsm, hsm_context ctx, const WatchEvent* evt )
246 {
247  // by default this function does nothing....
248  hsm_state ret=NULL;
249  switch (evt->evt) {
250  // but, when the 'toggle' button gets pressed....
251  case WATCH_TOGGLE_PRESSED:
252  // transition back to the stopped state
253  ret= StoppedState3();
254  break;
255  // also, when a 'tick' is sent, update our timer
256  case WATCH_TICK:
257  {
258  // our context is the watch object
259  Watch* watch=((WatchContext*)ctx)->watch;
260  // our event is the tick event
261  TickEvent* tick= (TickEvent*)evt;
262  // tick by this much time
263  TickTime ( watch, tick->time );
264  // print the total time
265  printf("%d,", watch->elapsed_time );
266  // indicate to the statemachine that we've take care of this event
267  // this stops the event from being sent to our parent
268  ret= HsmStateHandled();
269  }
270  break;
271  }
272  return ret;
273 }

299 int watch3region_events( int argc, char* argv[] )
300 {
301  Watch watch;
302  WatchContext ctx= { 0, &watch };
303  hsm_context_machine_t machine;
304  hsm_machine hsm= HsmMachineWithContext( &machine, &ctx.ctx );
305 
306  printf( "Stop Watch Sample with State Events.\n"
307  "Keys:\n"
308  "\t'1': reset button\n"
309  "\t'2': generic toggle button\n" );
310 
311  HsmStart( hsm, ActiveState3() );
312 
313  // while the statemachine is still running
314  while ( HsmIsRunning( hsm ) ) {
315  // get a key from the keyboard
316  int ch= PlatformGetKey();
317  // turn a '1' into the reset button,
318  // turn a '2' into the toggle button
319  WatchEvents events[]= { WATCH_RESET_PRESSED, WATCH_TOGGLE_PRESSED };
320  int index= ch-'1';
321  if ((index >=0) && (index < sizeof(events)/sizeof(WatchEvents))) {
322  // send the event to the state machine
323  // one of the handler functions will get called as a result
324  // ( for example: StoppedStateEvent. )
325  WatchEvent evt= { events[index] };
326  HsmProcessEvent( hsm, &evt );
327  printf(".");
328  }
329  else {
330  // send a tick to the machine
331  // we send this regardless of whether we think the watch is running or stopped.
332  // the logic of the statemachine internally knows which states are currently active
333  // and sends the event to the appropriate state. noting (in the code above) that only
334  // RunningStateEvent has code does anything when it hears the tick event.
335  TickEvent tick= { WATCH_TICK, 1 };
336  HsmProcessEvent( hsm, &tick.evt );
337  PlatformSleep(500);
338  }
339  };
340  return 0;
341 }
