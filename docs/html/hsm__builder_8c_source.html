115 static hsm_state HsbProcessEvent( hsm_machine hsm, hsm_context ctx, hsm_event evt )
116 {
117  // ways of getting the build_state_rec
118  // 1. override enter and use context [ would limit the depth of context stack; would limit mixing of user code ]
119  // 2. override enter and use a separate linked list [ would limit mixing ]
120  // 3. pointer math on current state.
121  hsm_state ret=NULL;
122  build_state_t* state= ( ((size_t)hsm->current) - offsetof( build_state_t, desc ) );
123  build_event_t* et;
124  // look through the built events
125  for (et= state->events; et; et=et->next) {
126  // determine if one matches
127  if (et->match && et->match( hsm, ctx, evt, et->user_data )) {
128  // run action(s)
129  build_action_t* at;
130  for (at= et->actions; at; at=at->next) {
131  at->run( hsm, ctx, evt );
132  }
133  // transition to target, or flag as handled.
134  ret= et->target ? et->target : HsmStateHandled();
135  break;
136  }
137  }
138  return ret;
139 }

268 static hsm_state BuildingEvent( hsm_machine hsm, hsm_context ctx, hsm_event evt )
269 {
270  hsm_state ret= HsmStateError(); // at the top level, by default, everythings an error.
271  builder_t * builder= ((builder_t*)ctx);
272  switch (evt->type) {
273  case _hsm_begin: {
274  build_state_t * state=evt->data.state;
275  if (state && !state->ctx.parent) {
276  ret= BuildingState();
277  }
278  else {
279  if (!state) {
280  Builder_Error( builder, "hsmBegin for unknown state." );
281  }
282  else
283  if (state->ctx.parent == &(state->ctx)) {
284  Builder_Error( builder, "state already finished building via hsmEnd.");
285  }
286  else {
287  Builder_Error( builder, "state already being built via hsmBegin.");
288  }
289  }
290  }
291  break;
292  default:
293  Builder_Error( builder, "invalid function for state" );
294  break;
295  }
296  return ret;
297 }

340 static hsm_state BuildingStateEvent( hsm_machine hsm, hsm_context ctx, hsm_event evt )
341 {
342  hsm_state ret= NULL;
343  builder_t * builder= ((builder_t*)ctx);
344  build_state_t* state= ((build_state_t*)builder->stack.context);
345  switch (evt->type)
346  {
347  case _hsm_enter:
348  {
349  if (!state->desc.enter && evt->data.enter) {
350  state->desc.enter= evt->data.enter;
351  ret= HsmStateHandled();
352  }
353  else {
354  if (state->desc.enter) {
355  Builder_Error( builder, "enter already specified." );
356  }
357  else
358  if (!evt->data.enter) {
359  Builder_Error( builder, "enter is null." );
360  }
361  ret= HsmStateError();
362  }
363  }
364  break;
365  case _hsm_on:
366  {
367  ret= BuildingOn();
368  }
369  break;
370  case _hsm_end: {
371  // an explict end finishes building the current state:
372  HsmContextPop( &(builder->stack) );
373  // 'self' used as a key to indicate end has been called:
374  state->ctx.parent= &(state->ctx);
375  // if this was the last matching begin/end pair, we're done:
376  if (builder->stack.count) {
377  ret= HsmStateHandled();
378  }
379  else {
380  ret= Building();
381  }
382  }
383  break;
384  }
385  return ret;
386 }

403 static hsm_state BuildingOnEvent( hsm_machine hsm, hsm_context ctx, hsm_event evt )
404 {
405  hsm_state ret=NULL;
406  builder_t* builder= ((builder_t*)ctx);
407  build_state_t* state= Builder_CurrentState( builder );
408  build_event_t* event= state->events;
409 
410  switch (evt->type) {
411  case _hsm_goto: {
412  // we can't lookup the actual states until 'complete'
413  // all we can do is record the structure for later
414  if (!event->target && evt->data.go) {
415  event->target= evt->data.go;
416  ret= HsmStateHandled();
417  }
418  else {
419  if (event->target) {
420  Builder_Error( builder,"goto already specified for this event");
421  }
422  else
423  if (!evt->data.go) {
424  Builder_Error( builder,"null target specified for goto");
425  }
426  ret= HsmStateError();
427  }
428  }
429  break;
430  case _hsm_run: {
431  build_action_t* action= BuildAction( event, evt->data.run );
432  if (action && action->run) {
433  ret= HsmStateHandled();
434  }
435  else {
436  if (!evt->data.run) {
437  Builder_Error( builder,"null action specified for run");
438  }
439  else
440  if (!action) {
441  Builder_Error( builder,"couldnt allocate action for run");
442  }
443  ret= HsmStateError();
444  }
445  }
446  break;
447  };
448  return ret;
449 }

485 int hsmState( const char * name )
486 {
487  int ret=0;
488  hsm_uint32 id;
489  hsm_state parent;
490  hash_entry_t* hash;

492 
493  if (!current) {
494  id= CRC32( name );

496  }
497  else {
498  id= CRC32_CAT( name, CRC32_CAT( "::", current->id ) );

501

503  if (hash) {
504  // TODO:

508  if (!hash->clientData) {
509  hash->clientData= BuildState( parent, name, id );
510  }
511  ret= id;
512  }
513  return ret;
514 }
