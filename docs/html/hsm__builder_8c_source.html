111 static hsm_state GenericEvent( hsm_status status )
112 {
113  // ways of getting the state_rec
114  // 1. override enter and use context [ would limit the depth of context stack; would limit mixing of user code ]
115  // 2. override enter and use a separate linked list [ would limit mixing ]
116  // 3. pointer math on current state.
117  hsm_state ret=NULL;
118  state_t* state= (state_t*) ( ((size_t)status->state) - offsetof( state_t, desc ) );
119  handler_t* et;
120  // look through the built event handlers
121  for (et= state->handlers; et; et=et->next) {
122  // determine if some guard blocks the event from running
123  guard_t* guard;
124  for (guard= &(et->guard); guard; guard=guard->next) {
125  if (!guard->match( status, guard->guard_data )) {
126  break;
127  }
128  }
129  // nope: no guard blocks this event from running
130  if (!guard) {
131  // run action(s)
132  action_t* at;
133  for (at= et->actions; at; at=at->next) {
134  at->run( status, at->action_data );
135  }
136  // transition to target, or flag as handled.
137  if (et->target) {
138  ret=et->target;
139  }
140  else {
141  ret= HsmStateHandled();
142  }
143  break;
144  }
145  }
146  return ret;
147 }

404 static hsm_state BuildingStateEvent( hsm_status status )
405 {
406  hsm_state ret= NULL;
407  builder_t * builder= ((builder_t*)status->ctx);
408  state_t* current= Builder_CurrentState( builder );
409  switch (status->evt->type)
410  {
411  case _hsm_enter:
412  {
413  if (!current->desc.enter && status->evt->data.enter) {
414  current->desc.enter= status->evt->data.enter;
415  ret= HsmStateHandled();
416  }
417  else {
418  if (current->desc.enter) {
419  Builder_Error( builder, "enter already specified." );
420  }
421  else
422  if (!status->evt->data.enter) {
423  Builder_Error( builder, "enter is null." );
424  }
425  ret= HsmStateError();
426  }
427  }
428  break; // a new event handler
429  case _hsm_guard:
430  {
431  ret= BuildingGuard();
432  }
433  break;
434  case _hsm_end: {
435  // an explict end finishes building the current state
436  hsm_context ctx= HsmContextPop( &(builder->stack) );
437  assert( ctx == &current->ctx && "we should have just popped current" );
438  // pointing to 'self' is used as a key to indicate end has been called
439  current->ctx.parent= &(current->ctx);
440  // setup the event handler, this is another key that the state is good to go.
441  current->desc.process= GenericEvent;
442  // if this was the last matching begin/end pair, we're done
443  if (builder->stack.count) {
444  ret= HsmStateHandled();
445  }
446  else {
447  ret= Building();
448  }
449  }
450  break;
451  }
452  return ret;
453 }

475 static hsm_state BuildingGuardEvent( hsm_status status )
476 {
477  hsm_state ret=NULL;
478  builder_t* builder= ((builder_t*)status->ctx);
479  state_t* state= Builder_CurrentState( builder );
480  assert(state);
481  if (state) {
482  handler_t* handler= state->handlers;
483 
484  switch (status->evt->type) {
485  case _hsm_goto: {
486  if (!handler->target && status->evt->data.go) {
487  handler->target= status->evt->data.go;
488  ret= HsmStateHandled();
489  }
490  else {
491  if (handler->target) {
492  Builder_Error( builder,"goto already specified for this event");
493  }
494  else
495  if (!status->evt->data.go) {
496  Builder_Error( builder,"unknown target specified for goto");
497  }
498  ret= HsmStateError();
499  }
500  }
501  break;
502  case _hsm_guard: {
503  const GuardEvent* guard_event= (const GuardEvent*)status->evt;
504  // on append, we want to add this guard to us
505  // otherwise we want to start a new event handler
506  if (guard_event->append) {
507  if (NewGuard( handler, guard_event->core.data.guard, guard_event->guard_data )) {
508  ret= HsmStateHandled();
509  }
510  else {
511  if (!guard_event->core.data.guard) {
512  Builder_Error( builder,"no guard specified");
513  }
514  else {
515  Builder_Error( builder,"couldnt allocate guard");
516  }
517  ret= HsmStateError();
518  }
519  }
520  }
521  break;
522  case _hsm_action: {
523  const ActionEvent* action_event= (const ActionEvent*)status->evt;
524  if (NewAction( handler, status->evt->data.run, action_event->action_data )) {
525  ret= HsmStateHandled();
526  }
527  else {
528  if (!status->evt->data.run) {
529  Builder_Error( builder,"no action specified for run");
530  }
531  else {
532  Builder_Error( builder,"couldnt allocate action for run");
533  }
534  ret= HsmStateError();
535  }
536  }
537  break;
538  };
539  }
540  return ret;
